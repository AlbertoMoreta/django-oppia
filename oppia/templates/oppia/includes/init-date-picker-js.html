function setUpDatePicker(){ startDate = $( "#id_start_date" ); startDate.datepicker({dateFormat: "yy-mm-dd", defaultDate: "-1m"}) .datepicker('setDate', startDate.val()); endDate = $( "#id_end_date" ); endDate.datepicker({dateFormat: "yy-mm-dd", defaultDate: "-1d"}) .datepicker('setDate', endDate.val()); }
